### 使用说明
```
    #用户名，密码，如1
	username = u"username"
	passwd = u"passwd"
	# cookies值得自己去找, 下面两个分别是深圳, 武昌,如3
	starts = u"%u6DF1%u5733%2CSZQ"
	ends = u"%u6B66%u660C%2CWCN"
	# 时间格式2018-02-11，自由选择
	dtime = u"2018-02-11"
	# 车次，选择第几趟，0则从上之下依次点击
	order = 0
	###乘客名，可多人，如[u"user1",u"user2"]
	users = [u"user1"]
	##席位
	xb = u"二等座"
	pz=u"成人票"
```
